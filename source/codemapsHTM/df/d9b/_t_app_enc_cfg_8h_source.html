607  template<typename T>
608  Void xReadStrToEnum(string in, std::vector<T> &val)
609  {
610  val.clear();
611 
612  char* cElement = NULL;
613  char* cString = new char[in.size()+1];
614  strcpy(cString, in.c_str());
615 
616  cElement = strtok( cString, " " );
617  while ( cElement != NULL )
618  {
619  T profile;
620  std::istringstream ss( cElement );
621  ss >> profile;
622  val.push_back( profile );
623  cElement = strtok( NULL, " " );
624  }
625  delete[] cString;
626  }
